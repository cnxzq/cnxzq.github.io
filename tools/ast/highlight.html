Identifier Highlight to spot everything

// Array shuffling code from Underscore.js.

_.shuffle = function(obj) {
    var shuffled = [], rand;
    each(obj, function(value, index, list) {
        rand = Math.floor(Math.random() * (index + 1));
        shuffled[index] = shuffled[rand];
        shuffled[rand] = value;
    });
    return shuffled;
};

When the cursor is placed on a variable or a function, all other identical references will be highlighted, particularly its declaration.
